2.微信分享信息配置

  /**
 * 配置微信分享信息
 * @param {Object} shareData 分享数据，不传仅展示分享按钮
 * @param {Boolean} shareWxWrok 处理微信企业号分享到企业微信
 * @{shareData} title  标题
 * @{shareData} desc  描述
 * @{shareData} imgSrc  lodosrc
 * @{shareData} shareUrl  分享的链接地址
 * @{shareData} type  分享类型,music、video或link，不填默认为link
 * @{shareData} dataUrl  如果type是music或video，则要提供数据链接，默认为空
 * @{shareData} success  接口调用成功时执行的回调函数
 * @{shareData} fail  接口调用失败时执行的回调函数
 * @{shareData} complete  接口调用完成时执行的回调函数，无论成功或失败都会执行
 * @{shareData} cancel  用户点击取消时的回调函数，仅部分有用户取消操作的api才会用到
 */
function wxShare(shareData, shareWxWrok) {   
   if (globalConfig.wechat.isWechatWork) {
        shareWxWrok = false;
    };
    !shareWxWrok && wx.showAllNonBaseMenuItem(shareData);
    if (shareData && !isEmptyObject(shareData)) {
        shareData.shareUrl = shareData.shareUrl || encodeURIComponent(location.href.split('#')[0]);
        let shareConfig = {
            title: shareData.title || '微商城', // 分享标题
            desc: shareData.desc, //描述
            link: shareData.shareUrl, // 分享链接
            imgUrl: shareData.imgUrl || "http://jkimg0626-10054369.cossh.myqcloud.com/jkjj/shareGift/img/logo2.png", // 分享图标
            type: shareData.type || '', // 分享类型,music、video或link，不填默认为link
            dataUrl: shareData.dataUrl || '', // 如果type是music或video，则要提供数据链接，默认为空
            success: shareData.success, //接口调用成功时执行的回调函数
            cancel: shareData.cancel, //用户点击取消时的回调函数，仅部分有用户取消操作的api才会用到
            fail: shareData.fail, //接口调用失败时执行的回调函数
            complete: shareData.complete //接口调用完成时执行的回调函数，无论成功或失败都会执行
        }
        wx.onMenuShareTimeline(shareConfig); //分享朋友圈
        wx.onMenuShareAppMessage(shareConfig); //分享给朋友
        wx.onMenuShareQQ(shareConfig); //分享到QQ
        wx.onMenuShareWeibo(shareConfig); //享到腾讯微博
        wx.onMenuShareQZone(shareConfig); //分享到QQ空间
        wx.onMenuShareWechat && wx.onMenuShareWechat(shareConfig); //从企业微信分享到微信
    }
}
